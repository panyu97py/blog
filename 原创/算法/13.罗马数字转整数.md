```javascript
const I = 1
const V = 5
const X = 10
const L = 50
const C = 100
const D = 500
const M = 1000
const IV = 4
const IX = 9
const XL = 40
const XC = 90
const CD = 400
const CM = 900

const SIMPLE_ROMAN_MAP = new Map(Object.entries({I, V, X, L, C, D, M}))

const COMPLEX_ROMAN_MAP = new Map(Object.entries({IV, IX, XL, XC, CD, CM}))

/**
 * @param {string} s
 * @return {number}
 */
var romanToInt = function (s) {
    let val = 0
    let i = 0
    while (i < s.length) {
        const x = s[i]
        const y = s[i] + s[i + 1]
        if (COMPLEX_ROMAN_MAP.has(y)) {
            val += COMPLEX_ROMAN_MAP.get(y)
            i += 2
            continue
        }
        val += SIMPLE_ROMAN_MAP.get(x)
        i += 1
    }
    return val
};
```
